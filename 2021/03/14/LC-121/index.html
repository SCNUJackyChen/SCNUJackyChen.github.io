class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int n = prices.size();
        vector<int> f(n + 1);
        for(int i = 1, mini = INT_MAX; i <= n; i++) {
            f[i] = max(f[i - 1], prices[i - 1] - mini);
            mini = min(mini, prices[i - 1]);
        }

        int res = 0;
        for(int i = n, maxi = 0; i; i--) {
            res = max(res, maxi - prices[i - 1] + f[i - 1]);
            maxi = max(maxi, prices[i - 1]);
        }
        return res;
    }
};

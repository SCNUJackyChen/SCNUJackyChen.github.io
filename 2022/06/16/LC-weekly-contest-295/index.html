class Solution {
public:
    int totalSteps(vector<int>& nums) {
        stack<pair<int, int>> stk;
        int res = 0;
        for (int x : nums) {
            int ma = 0;
            while (stk.size() && stk.top().first <= x) {
                ma = max(ma, stk.top().second);
                stk.pop();
            }
            if (stk.size()) ma ++;
            res = max(res, ma);
            stk.push({x, ma});
        }
        return res;
    }
};

class Solution {
public:
    int minimumObstacles(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> dist(n, vector<int>(m, -1));

        deque<pair<int, int>> dq;
        dq.push_back({0, 0});
        dist[0][0] = grid[0][0] == 0 ? 0 : 1;

        int dx[4] = {-1, 0, 1, 0}, dy[4] = {0, 1, 0, -1};

        while (dq.size()) {
            auto top = dq.front(); dq.pop_front();
            for (int i = 0; i < 4; i++) {
                int a = top.first + dx[i], b = top.second + dy[i];
                if (a >= 0 && a < n && b >= 0 && b < m) {
                    if (dist[a][b] != -1) continue;
                    if (grid[a][b] == 0) {
                        dist[a][b] = dist[top.first][top.second];
                        dq.push_front({a, b});
                    } else {
                        dist[a][b] = dist[top.first][top.second] + 1;
                        dq.push_back({a, b});
                    }
                }
            }
        }
        return dist[n - 1][m - 1];
    }
};
